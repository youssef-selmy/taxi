        border: Border.all(color: context.colors.outline),
        borderRadius: BorderRadius.circular(24),
      ),
      child: Column(
        children: [
          Padding(
            padding: const EdgeInsets.symmetric(vertical: 8, horizontal: 16),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              children: <Widget>[
                Text('Manage Your Contact', style: context.textTheme.titleSmall),
                Row(
                  children: [
                    SizedBox(
                      width: 270,
                      child: AppTextField(
                        fillColor: context.colors.surface,
                        density: TextFieldDensity.noDense,
                        hint: 'Search',
                        prefixIcon: Icon(
                          BetterIcons.search01Filled,
                          color: context.colors.onSurfaceVariant,
                        ),
                      ),
                    ),
                  ],
                ),
              ],
            ),
          ),
          AppDivider(height: 20),
          SizedBox(height: 16),
          SizedBox(
            height: 400,
            child: AppDataTable<List<_Contact>, dynamic>(
              minWidth: 1400,
              data: data,
              getPageInfo:
                  (_) => OffsetPageInfo(
                    hasNextPage: false,
                    hasPreviousPage: false,
                  ),
              getRowCount: (list) => list.length,
              onPageChanged: (_) {},
              paging: OffsetPaging(limit: 10, offset: 0),
              columns: [
                DataColumn2(
                  fixedWidth: 60,
                  label: const AppCheckbox(value: false),
                ),
                _buildSortableColumn(context, label: 'Name', width: 220),
                _buildSortableColumn(context, label: 'Company', width: 215),
                _buildSortableColumn(context, label: 'Value', width: 140),
                _buildSortableColumn(context, label: 'Cause', width: 180),
                _buildSortableColumn(context, label: 'Status', width: 150),
                DataColumn2(fixedWidth: 90, label: const Text('')),
              ],
              rowBuilder: (list, index) {
                final contact = list[index];
                return DataRow(
                  onSelectChanged: (_) {},
                  cells: [
                    DataCell(AppCheckbox(value: false)),
                    DataCell(
                      Column(
                        crossAxisAlignment: CrossAxisAlignment.start,
                        mainAxisAlignment: MainAxisAlignment.center,
                        spacing: 4,
                        children: [
                          Text(
                            contact.name,
                            style: context.textTheme.labelLarge,
                          ),
                          Text(
                            contact.email,
                            style: context.textTheme.bodySmall?.variant(
                              context,
                            ),
                          ),
                        ],
                      ),
                    ),
                    DataCell(
                      Text(
                        contact.company,
                        style: context.textTheme.labelLarge,
                      ),
                    ),
                    DataCell(
                      Column(
                        crossAxisAlignment: CrossAxisAlignment.start,
                        mainAxisAlignment: MainAxisAlignment.center,
                        spacing: 4,
                        children: [
                          Text(
                            contact.value,
                            style: context.textTheme.labelLarge,
                          ),
                          Text(
                            contact.currency,
                            style: context.textTheme.bodySmall?.variant(
                              context,
                            ),
                          ),
                        ],
                      ),
                    ),
                    DataCell(
                      Text(contact.cause, style: context.textTheme.labelLarge),
                    ),
                    DataCell(_buildStatus(contact.status)),
                    DataCell(
                      AppIconButton(
                        icon: BetterIcons.moreVerticalCircle01Outline,
                        isCircular: true,
                      ),
                    ),
                  ],
                );
              },
            ),
          ),
        ],
      ),
    );
  }

  Widget _buildSortIcon(BuildContext context) {
    return Icon(
      BetterIcons.unfoldMoreFilled,
      color: context.colors.onSurfaceVariantLow,
      size: 20,
    );
  }

  Widget _buildStatus(_ContactStatus status) {
    return AppTag(
      prefixIcon: switch (status) {
        _ContactStatus.paid => BetterIcons.checkmarkCircle02Filled,
        _ContactStatus.waiting => BetterIcons.cancelCircleFilled,
      },
      text: switch (status) {
        _ContactStatus.paid => 'Paid',
        _ContactStatus.waiting => 'Waiting',
      },
      color: switch (status) {
        _ContactStatus.paid => SemanticColor.success,
        _ContactStatus.waiting => SemanticColor.warning,
      },
    );
  }

  DataColumn2 _buildSortableColumn(
    BuildContext context, {
    required String label,
    required double width,
  }) {
    return DataColumn2(
      fixedWidth: width,
      label: Row(
        children: [
          Text(label, style: context.textTheme.bodyMedium),
          const SizedBox(width: 4),
          _buildSortIcon(context),
        ],
      ),
    );
  }

  Widget _buildInfoRow(
    BuildContext context, {
    required String label,
    required Widget value,
  }) {
    return Row(
      mainAxisAlignment: MainAxisAlignment.spaceBetween,
      children: <Widget>[
        Text(label, style: context.textTheme.bodyMedium?.variant(context)),
        value,
      ],
    );
  }
}
